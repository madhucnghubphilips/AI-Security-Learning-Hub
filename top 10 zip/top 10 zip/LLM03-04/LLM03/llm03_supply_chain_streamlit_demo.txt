files["controls.py"] = r'''
import re
from datetime import datetime
APPROVED_DEPENDENCIES = {
"ai-security-helper": ["1.0.0"],
}
KNOWN_BAD_PHRASES = [
"share admin credentials",
"send logs to external",
"disable authentication",
"hardcode token",
"place admin credentials in application logs",
]
def validate_dependency(name: str, version: str) -> dict:
allowed_versions = APPROVED_DEPENDENCIES.get(name, [])
if version not in allowed_versions:
return {
"allowed": False,
"dependency": name,
"version": version,
"reason": f"Blocked dependency {name}=={version}. Version is not in approved allowlist.",
}
return {
"allowed": True,
"dependency": name,
"version": version,
"reason": "Dependency version approved.",
}
def scan_prompt_template(prompt_template: str) -> dict:
lower_prompt = prompt_template.lower()
for phrase in KNOWN_BAD_PHRASES:
if phrase in lower_prompt:
return {
"allowed": False,
"reason": f"Prompt template blocked. Suspicious instruction detected: '{phrase}'.",
}
return {
"allowed": True,
"reason": "Prompt template passed integrity check.",
}
def validate_rag_context(context: str) -> dict:
lower_context = context.lower()
for phrase in KNOWN_BAD_PHRASES:
if phrase in lower_context:
return {
"allowed": False,
"reason": f"RAG context blocked. Untrusted or unsafe content detected: '{phrase}'.",
}
return {
"allowed": True,
"reason": "RAG context passed validation.",
}
def enforce_output_policy(response: str) -> dict:
lower_response = response.lower()
for phrase in KNOWN_BAD_PHRASES:
if phrase in lower_response:
return {
"allowed": False,
"reason": f"Output blocked. Unsafe recommendation detected: '{phrase}'.",
}
return {
"allowed": True,
"reason": "Output passed policy enforcement.",
}
